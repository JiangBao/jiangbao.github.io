function FileListPlugin(options={}) {
  this.filename = options.filename ? options.filename : 'filelist.md';
}

  // æ³¨å†Œè¾“å‡ºæ–‡ä»¶ä¹‹å‰çš„é’©å­ emit

      filelist += `* ${filename}\n`;

    // å°†åˆ—è¡¨ä½œä¸ºæ–°çš„æ–‡ä»¶èµ„æºï¼Œæ’å…¥åˆ° webpack æ„å»º

      }
    }

    cb();
  });
}

module.exports = FileListPlugin;
